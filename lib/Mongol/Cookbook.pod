=head2 Custom ids

	package Models::User {
		use Moose;

		extends 'Mongol::Model';

		use UUID qw( uuid );

		with 'Mongol::Roles::Core';

		has '+id' => (
			default => sub { uuid() },
		);

		has 'username' => (
			is => 'ro',
			isa => 'Str',
			required => 1,
		);

		has 'password' => (
			is => 'ro',
			isa => 'Str',
			required => 1,
		);

		...

		__PACKAGE__->meta()->make_immutable();
	}

	1;
